# -----------------------------------------------------------------------------
# Core Numerical Computing Stack
# -----------------------------------------------------------------------------

# NumPy: Foundation for high-performance array computing and linear algebra
# - Used for: Dense correlation matrix computation, eigendecomposition preprocessing,
#   vectorized log-return calculations, Z-score normalization, memory-mapped arrays
# - Version Rationale: 1.24.3 provides stable ABI, optimized BLAS/LAPACK bindings,
#   and critical fixes for complex eigenvalue algorithms used in PCA-whitening
# - Size: ~50 MB (includes OpenBLAS on most platforms)
numpy==1.24.3

# SciPy: Advanced scientific computing and statistical analysis
# - Used for: Spectral decomposition (scipy.linalg.eigh), Maximum Likelihood 
#   Estimation (scipy.optimize.minimize with L-BFGS-B), Generalized Pareto 
#   Distribution fitting (scipy.stats), FFT-based autocovariance (scipy.fft)
# - Version Rationale: 1.10.1 includes optimized QR eigenvalue algorithms,
#   robust optimization convergence for constrained GPD parameter estimation,
#   and vectorized statistical distribution functions
# - Dependencies: Requires NumPy >= 1.19.5 (satisfied by numpy==1.24.3)
# - Size: ~35 MB
scipy==1.10.1

# Pandas: High-performance dataframe operations for financial time series
# - Used for: TAQ data ingestion, asynchronous tick-data alignment, forward-fill
#   imputation, intraday seasonality profiling, rolling-window quantile computation
# - Version Rationale: 2.0.2 introduces Arrow-backed memory efficiency (reducing
#   RAM footprint by ~30% for large TAQ datasets), copy-on-write semantics 
#   (eliminating hidden data duplication), and 10x faster rolling operations
# - Dependencies: Requires NumPy >= 1.20.3, Python >= 3.8
# - Size: ~40 MB
pandas==2.0.2

# psutil: Cross-platform system and process monitoring
# - Used for: Real-time memory profiling to prevent OOM errors during dense
#   correlation matrix computation, CPU utilization tracking for parallel MLE
#   optimization, disk I/O monitoring for memory-mapped array operations
# - Version Rationale: 5.9.5 provides accurate resident set size (RSS) measurement
#   critical for managing 2.6+ billion data point workflows, and fixes race 
#   conditions in multi-threaded environments
# - Dependencies: Pure Python with C extensions (no external dependencies)
# - Size: ~500 KB
psutil==5.9.5
